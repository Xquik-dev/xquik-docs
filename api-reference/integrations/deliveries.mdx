title: "List Integration Deliveries"
description: "Retrieve delivery history for a specific integration"
api: "GET /integrations/{id}/deliveries"
keywords: ["integration", "deliveries", "history", "status", "attempts"]
---

<Info>
  This endpoint does not count against your monthly usage quota.
</Info>

## Path parameters

<ParamField path="id" type="string" required>
  The unique integration ID.
</ParamField>

## Headers

<ParamField header="x-api-key" type="string" required>
  Your API key. Session cookie authentication is also supported. Generate a key from the [dashboard](https://xquik.com/dashboard).
</ParamField>

## Query parameters

<ParamField query="limit" type="number">
  Maximum number of deliveries to return. Defaults to 100.
</ParamField>

## Response

<Tabs>
  <Tab title="200 OK">
    <ResponseField name="deliveries" type="object[]">Array of delivery objects, newest first.</ResponseField>
    <ResponseField name="deliveries[].id" type="string">Unique delivery ID.</ResponseField>
    <ResponseField name="deliveries[].eventType" type="string">Event type that triggered the delivery.</ResponseField>
    <ResponseField name="deliveries[].status" type="string">Delivery status: `"delivered"`, `"failed"`, `"pending"`, or `"exhausted"`.</ResponseField>
    <ResponseField name="deliveries[].attempts" type="number">Number of delivery attempts made.</ResponseField>
    <ResponseField name="deliveries[].deliveredAt" type="string">ISO 8601 timestamp of successful delivery. `null` if not yet delivered.</ResponseField>
    <ResponseField name="deliveries[].lastError" type="string">Error message from the last failed attempt. `null` if no errors.</ResponseField>
    <ResponseField name="deliveries[].lastStatusCode" type="number">HTTP status code from the last attempt. `null` if not applicable.</ResponseField>
    <ResponseField name="deliveries[].sourceId" type="string">ID of the source object that triggered the event.</ResponseField>
    <ResponseField name="deliveries[].sourceType" type="string">Type of the source object (e.g. `"monitor"`).</ResponseField>
    <ResponseField name="deliveries[].createdAt" type="string">ISO 8601 timestamp of when the delivery was created.</ResponseField>
