
## Path parameters

<ParamField path="id" type="string" required>
  The unique integration ID. Returned when you [create an integration](/api-reference/integrations/create) or [list integrations](/api-reference/integrations/list).
</ParamField>

## Headers

<ParamField header="x-api-key" type="string" required>
  Your API key. Session cookie authentication is also supported. Generate a key from the [dashboard](https://xquik.com/dashboard).
</ParamField>

## Response

<Tabs>
  <Tab title="200 OK">
    <ResponseField name="id" type="string">Unique integration ID.</ResponseField>
    <ResponseField name="type" type="string">Integration type (e.g. `"telegram"`).</ResponseField>
    <ResponseField name="name" type="string">Integration name.</ResponseField>
    <ResponseField name="config" type="object">Integration configuration.</ResponseField>
    <ResponseField name="eventTypes" type="string[]">Subscribed event types.</ResponseField>
    <ResponseField name="messageTemplate" type="string">Custom message template for notifications. `null` if not set.</ResponseField>
    <ResponseField name="scopeAllMonitors" type="boolean">Whether the integration receives events from all monitors.</ResponseField>
    <ResponseField name="silentPush" type="boolean">Whether Telegram notifications are sent silently.</ResponseField>
    <ResponseField name="filters" type="object">Event filtering rules. `null` if not set.</ResponseField>
    <ResponseField name="isActive" type="boolean">Whether the integration is active.</ResponseField>
    <ResponseField name="createdAt" type="string">ISO 8601 creation timestamp.</ResponseField>
    <ResponseField name="updatedAt" type="string">ISO 8601 timestamp of last update.</ResponseField>

    ```json
    {
      "id": "4",
      "type": "telegram",
      "name": "My Telegram Bot",
      "config": { "chatId": "-1001234567890" },
      "eventTypes": ["tweet.new", "tweet.reply"],
      "messageTemplate": null,
      "scopeAllMonitors": true,
      "silentPush": false,
      "filters": null,
      "isActive": true,
      "createdAt": "2026-03-01T09:00:00.000Z",
      "updatedAt": "2026-03-01T09:00:00.000Z"
    }
    ```
  </Tab>
  <Tab title="400 Invalid ID">
    ```json
    { "error": "invalid_id", "message": "Invalid integration ID format" }
    ```
    The provided integration ID is not a valid format.
  </Tab>
  <Tab title="401 Unauthenticated">
    ```json
    { "error": "unauthenticated", "message": "Missing or invalid API key" }
    ```
    Missing or invalid API key.
  </Tab>
  <Tab title="404 Not Found">
    ```json
    { "error": "not_found", "message": "Integration not found" }
    ```
    No integration exists with this ID, or it belongs to a different account.
  </Tab>
</Tabs>
