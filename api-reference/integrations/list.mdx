## Headers

<ParamField header="x-api-key" type="string" required>
  Your API key. Session cookie authentication is also supported. Generate a key from the [dashboard](https://xquik.com/dashboard).
</ParamField>

## Response

<Tabs>
  <Tab title="200 OK">
    <ResponseField name="integrations" type="object[]">Array of integration objects.</ResponseField>
    <ResponseField name="integrations[].id" type="string">Unique integration ID.</ResponseField>
    <ResponseField name="integrations[].type" type="string">Integration type (e.g. `"telegram"`).</ResponseField>
    <ResponseField name="integrations[].name" type="string">Integration name.</ResponseField>
    <ResponseField name="integrations[].config" type="object">Integration configuration.</ResponseField>
    <ResponseField name="integrations[].eventTypes" type="string[]">Subscribed event types.</ResponseField>
    <ResponseField name="integrations[].messageTemplate" type="string">Custom message template for notifications. `null` if not set.</ResponseField>
    <ResponseField name="integrations[].scopeAllMonitors" type="boolean">Whether the integration receives events from all monitors.</ResponseField>
    <ResponseField name="integrations[].silentPush" type="boolean">Whether Telegram notifications are sent silently.</ResponseField>
    <ResponseField name="integrations[].filters" type="object">Event filtering rules. `null` if not set.</ResponseField>
    <ResponseField name="integrations[].isActive" type="boolean">Whether the integration is active.</ResponseField>
    <ResponseField name="integrations[].createdAt" type="string">ISO 8601 creation timestamp.</ResponseField>
    <ResponseField name="integrations[].updatedAt" type="string">ISO 8601 timestamp of last update.</ResponseField>
