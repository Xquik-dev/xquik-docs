## Path Parameters

<ParamField path="id" type="string" required>
  The event ID to retrieve.
</ParamField>

## Headers

<ParamField header="x-api-key" type="string" required>
  Your API key.
</ParamField>

## Response

<Tabs>
  <Tab title="200 OK">
    <ResponseField name="id" type="string">
      Unique event identifier.
    </ResponseField>
    <ResponseField name="type" type="string">
      Event type (e.g. `tweet.new`, `tweet.reply`, `tweet.quote`, `tweet.retweet`).
    </ResponseField>
    <ResponseField name="monitorId" type="string">
      ID of the monitor that captured this event.
    </ResponseField>
    <ResponseField name="username" type="string">
      X username that triggered the event.
    </ResponseField>
    <ResponseField name="occurredAt" type="string">
      ISO 8601 timestamp of when the event occurred on X.
    </ResponseField>
    <ResponseField name="data" type="object">
      Event-specific payload.
      <Expandable title="data object">
        <ResponseField name="tweetId" type="string">
          X tweet ID.
        </ResponseField>
        <ResponseField name="text" type="string">
          Tweet text content.
        </ResponseField>
        <ResponseField name="metrics" type="object">
          Engagement metrics at capture time.
          <Expandable title="metrics object">
            <ResponseField name="likes" type="number">
              Like count.
            </ResponseField>
            <ResponseField name="retweets" type="number">
              Retweet count.
            </ResponseField>
            <ResponseField name="replies" type="number">
              Reply count.
            </ResponseField>
          </Expandable>
        </ResponseField>
      </Expandable>
    </ResponseField>
    <ResponseField name="xEventId" type="string">
      X platform event ID. Present only for certain event types.
    </ResponseField>

    ```json
    {
      "id": "9001",
      "type": "tweet.new",
      "monitorId": "7",
      "username": "elonmusk",
      "occurredAt": "2026-02-24T14:22:00.000Z",
      "data": {
        "tweetId": "1893456789012345678",
        "text": "The future is now.",
        "metrics": {
          "likes": 42000,
          "retweets": 8500,
          "replies": 3200
        }
      },
      "xEventId": "evt_abc123"
    }
    ```
  </Tab>
  <Tab title="401 Unauthenticated">
    ```json
    { "error": "unauthenticated" }
    ```
    Missing or invalid API key. Check the `x-api-key` header value.
  </Tab>
  <Tab title="400 Invalid ID">
    ```json
    { "error": "invalid_id" }
    ```
    The provided event ID is not a valid format.
  </Tab>
  <Tab title="404 Not Found">
    ```json
    { "error": "not_found" }
    ```
    No event exists with this ID, or it belongs to a different account.
  </Tab>
</Tabs>
