  ```go Go
  package main

  import (
      "bytes"
      "encoding/json"
      "fmt"
      "net/http"

  func main() {
      ticketID := "tkt_a1b2c3d4e5f6a1b2c3d4e5f6"
      body, _ := json.Marshal(map[string]interface{}{
          "body": "That worked, thank you!",
      })

      req, err := http.NewRequest("POST", "https://xquik.com/api/v1/support/tickets/"+ticketID+"/messages", bytes.NewReader(body))
      if err != nil {
          panic(err)
      }
      req.Header.Set("x-api-key", "xq_YOUR_KEY_HERE")
      req.Header.Set("Content-Type", "application/json")

      resp, err := http.DefaultClient.Do(req)
      if err != nil {
          panic(err)
      }
      defer resp.Body.Close()

      var data map[string]interface{}
      if err := json.NewDecoder(resp.Body).Decode(&data); err != nil {
          panic(err)
      }
      fmt.Println(data)
  }
  ```

## Path parameters

<ParamField path="id" type="string" required>
  The ticket public ID (e.g. `tkt_a1b2c3d4e5f6a1b2c3d4e5f6`). Returned when you [create a ticket](/api-reference/support/create) or [list tickets](/api-reference/support/list).
</ParamField>

## Headers

<ParamField header="x-api-key" type="string" required>
  Your API key. Session cookie authentication is also supported. Generate a key from the [dashboard](https://xquik.com/dashboard).
</ParamField>

<ParamField header="Content-Type" type="string" required>
  Must be `application/json`.
</ParamField>

## Body

<ParamField body="body" type="string" required>
  Message content. 1-10,000 characters.
</ParamField>

## Response

<Tabs>
  <Tab title="201 Created">
    <ResponseField name="publicId" type="string">Ticket public ID the message was added to.</ResponseField>

    ```json
    {
      "publicId": "tkt_a1b2c3d4e5f6a1b2c3d4e5f6"
    }
    ```
  </Tab>
  <Tab title="400 Invalid Input">
    ```json
    { "error": "invalid_input", "message": "Invalid input. Check the request body." }
    ```
    Missing or invalid `body` field.
  </Tab>
  <Tab title="401 Unauthenticated">
    ```json
    { "error": "unauthenticated", "message": "Authentication required. Provide a valid API key or bearer token." }
    ```
    Missing or invalid API key.
  </Tab>
  <Tab title="404 Not Found">
    ```json
    { "error": "not_found", "message": "Ticket not found" }
    ```
    No ticket exists with this ID, or it belongs to a different account.
  </Tab>
</Tabs>
