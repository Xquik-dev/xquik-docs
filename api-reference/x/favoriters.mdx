  ```javascript Node.js
  const tweetId = "1893456789012345678";
  const response = await fetch(`https://xquik.com/api/v1/x/tweets/${tweetId}/favoriters`, {
  ```

  ```python Python
  import requests

  response = requests.get(

  )
  data = response.json()
  print(data)
  ```

## Path parameters

<ParamField path="id" type="string" required>
  Tweet ID (numeric string).
</ParamField>

## Headers

<ParamField header="x-api-key" type="string" required>
  Your API key. Session cookie authentication is also supported.
</ParamField>

## Response

<Tabs>
  <Tab title="200 OK">
    <ResponseField name="users" type="object[]">
      Array of users who liked the tweet.
      <Expandable title="user object">
        <ResponseField name="id" type="string">User ID.</ResponseField>
        <ResponseField name="username" type="string">X username.</ResponseField>
        <ResponseField name="name" type="string">Display name.</ResponseField>
        <ResponseField name="followers" type="number">Follower count. Omitted if unavailable.</ResponseField>
        <ResponseField name="verified" type="boolean">Whether the user is verified. Omitted if unavailable.</ResponseField>
        <ResponseField name="profilePicture" type="string">Profile picture URL. Omitted if unavailable.</ResponseField>
      </Expandable>
    </ResponseField>
    <ResponseField name="count" type="number">Number of users returned.</ResponseField>

    ```json
    {
      "users": [
        {
          "id": "987654321",
          "username": "xquikcom",
          "name": "Xquik",
          "followers": 10000,
          "verified": true
        }
      ],
      "count": 1
    }
    ```
  </Tab>
  <Tab title="400 Invalid tweet ID">
    ```json
    { "error": "invalid_tweet_id" }
    ```
    The tweet ID is empty or invalid.
  </Tab>
  <Tab title="401 Unauthenticated">
    ```json
    { "error": "unauthenticated" }
    ```
    Missing or invalid API key.
  </Tab>
  <Tab title="402 Subscription required">
    ```json
    { "error": "no_subscription" }
    ```
    No active subscription or usage limit reached.
  </Tab>
  <Tab title="502 X API unavailable">
    ```json
    { "error": "x_api_unavailable" }
    ```
    The X data source returned an error. Retry after a short delay.
  </Tab>
</Tabs>
