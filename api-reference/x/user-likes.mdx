  ```bash cURL
  curl https://xquik.com/api/v1/x/users/44196397/likes \
    -H "x-api-key: xq_YOUR_KEY_HERE" | jq

  # Page 2
  curl -G https://xquik.com/api/v1/x/users/44196397/likes \
    --data-urlencode "cursor=abc123" \
    -H "x-api-key: xq_YOUR_KEY_HERE" | jq
  ```

  ```javascript Node.js
  const userId = "44196397";
  const response = await fetch(`https://xquik.com/api/v1/x/users/${userId}/likes`, {
    headers: { "x-api-key": "xq_YOUR_KEY_HERE" },
  });
  const data = await response.json();
  console.log(data.tweets);

  // Paginate
  if (data.has_next_page) {
    const next = await fetch(
      `https://xquik.com/api/v1/x/users/${userId}/likes?cursor=${data.next_cursor}`,
      { headers: { "x-api-key": "xq_YOUR_KEY_HERE" } },
    );
    console.log((await next.json()).tweets);
  }
  ```

  ```python Python
  import requests

  response = requests.get(
      "https://xquik.com/api/v1/x/users/44196397/likes",
      headers={"x-api-key": "xq_YOUR_KEY_HERE"},
  )
  data = response.json()
  print(data["tweets"])

  # Paginate
  while data["has_next_page"]:
      data = requests.get(
          "https://xquik.com/api/v1/x/users/44196397/likes",
          params={"cursor": data["next_cursor"]},
          headers={"x-api-key": "xq_YOUR_KEY_HERE"},
      ).json()
      print(data["tweets"])
  ```
</CodeGroup>

## Path parameters

<ParamField path="id" type="string" required>
  X user ID (numeric string) or username.
</ParamField>

## Query parameters

<ParamField query="cursor" type="string">
  Pagination cursor. Pass the `next_cursor` value from the previous response to fetch the next page.
</ParamField>

## Headers

<ParamField header="x-api-key" type="string" required>
  Your API key. Session cookie authentication is also supported.
</ParamField>
