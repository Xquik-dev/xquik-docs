  ```javascript Node.js
  const params = new URLSearchParams({ q: "from:xquikcom giveaway" });
  const response = await fetch(`https://xquik.com/api/v1/x/tweets/search?${params}`, {
    headers: { "x-api-key": "xq_YOUR_KEY_HERE" },
  });
  const data = await response.json();
  console.log(data.tweets);

  // Paginate through all results
  let cursor = data.next_cursor;
  while (data.has_next_page) {
    const next = await fetch(
      `https://xquik.com/api/v1/x/tweets/search?${new URLSearchParams({ q: "from:xquikcom giveaway", cursor })}`,
      { headers: { "x-api-key": "xq_YOUR_KEY_HERE" } },
    );
    const page = await next.json();
    console.log(page.tweets);
    if (!page.has_next_page) break;
    cursor = page.next_cursor;
  }
  ```

## Query parameters

<ParamField query="q" type="string" required>
  Search query. Supports X search operators such as `from:username`, `to:username`, `#hashtag`, and boolean operators.
</ParamField>

<ParamField query="queryType" type="string">
  Sort order for search results. `Top` returns most relevant tweets, `Latest` returns most recent. Defaults to `Latest`.
</ParamField>

<ParamField query="cursor" type="string">
  Pagination cursor. Pass the `next_cursor` value from the previous response to fetch the next page.
</ParamField>

<ParamField query="sinceTime" type="string">
  Filter tweets published after this date. Uses X search `since:` operator format (e.g. `2026-03-01`).
</ParamField>

<ParamField query="untilTime" type="string">
  Filter tweets published before this date. Uses X search `until:` operator format (e.g. `2026-03-27`).
</ParamField>

<ParamField query="limit" type="integer" deprecated>
  Deprecated. Use cursor-based pagination instead. Previously set a maximum result count. The server now returns a `X-Deprecation` header when this parameter is present.
</ParamField>

## Headers

<ParamField header="x-api-key" type="string" required>
  Your API key. Session cookie authentication is also supported.
</ParamField>
