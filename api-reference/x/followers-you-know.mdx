  ```bash cURL
  curl https://xquik.com/api/v1/x/users/44196397/followers-you-know \
    -H "x-api-key: xq_YOUR_KEY_HERE" | jq

  # Page 2
  curl -G https://xquik.com/api/v1/x/users/44196397/followers-you-know \
  ```

  ```javascript Node.js
  const userId = "44196397";
  const response = await fetch(`https://xquik.com/api/v1/x/users/${userId}/followers-you-know`, {
    headers: { "x-api-key": "xq_YOUR_KEY_HERE" },
  });
  const data = await response.json();
  console.log(data.users);

  // Paginate
  if (data.has_next_page) {
    const next = await fetch(
      `https://xquik.com/api/v1/x/users/${userId}/followers-you-know?cursor=${data.next_cursor}`,
      { headers: { "x-api-key": "xq_YOUR_KEY_HERE" } },
    );
    console.log((await next.json()).users);
  }
  ```

  ```python Python
  import requests

  response = requests.get(
      "https://xquik.com/api/v1/x/users/44196397/followers-you-know",
      headers={"x-api-key": "xq_YOUR_KEY_HERE"},
  )
  data = response.json()
  print(data["users"])

  # Paginate
  while data["has_next_page"]:
      data = requests.get(
          "https://xquik.com/api/v1/x/users/44196397/followers-you-know",
          params={"cursor": data["next_cursor"]},
          headers={"x-api-key": "xq_YOUR_KEY_HERE"},
      ).json()
      print(data["users"])
  ```

## Path parameters

<ParamField path="id" type="string" required>
  X user ID (numeric string).
</ParamField>

## Query parameters

<ParamField query="cursor" type="string">
  Pagination cursor. Pass the `next_cursor` value from the previous response to fetch the next page.
</ParamField>

## Headers

<ParamField header="x-api-key" type="string" required>
  Your API key. Session cookie authentication is also supported.
</ParamField>

## Response

<Tabs>
  <Tab title="200 OK">
    <ResponseField name="users" type="object[]">
      Array of mutual follower profiles.
      <Expandable title="user object">
        <ResponseField name="id" type="string">User ID.</ResponseField>
        <ResponseField name="username" type="string">X username.</ResponseField>

    <ResponseField name="has_next_page" type="boolean">
      Whether more results are available.
    </ResponseField>
    <ResponseField name="next_cursor" type="string">
      Opaque cursor for the next page. Empty string when no more results.
    </ResponseField>
