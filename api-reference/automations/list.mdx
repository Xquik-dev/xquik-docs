## Headers

<ParamField header="x-api-key" type="string" required>
  Your API key. Session cookie authentication is also supported. Generate a key from the [dashboard](https://xquik.com/dashboard).
</ParamField>

## Response

<Tabs>
  <Tab title="200 OK">
    <ResponseField name="items" type="object[]">Array of flow objects.</ResponseField>
    <ResponseField name="items[].id" type="string">Unique flow ID.</ResponseField>
    <ResponseField name="items[].name" type="string">Flow name.</ResponseField>
    <ResponseField name="items[].slug" type="string">URL-safe flow identifier.</ResponseField>
    <ResponseField name="items[].triggerType" type="string">Trigger type: `monitor_event`, `schedule`, `search`, or `webhook_inbound`.</ResponseField>
    <ResponseField name="items[].triggerConfig" type="object">Trigger-specific configuration.</ResponseField>
    <ResponseField name="items[].isActive" type="boolean">Whether the flow is currently active.</ResponseField>
    <ResponseField name="items[].runCount" type="string">Total number of executions.</ResponseField>
    <ResponseField name="items[].lastRunAt" type="string | null">ISO 8601 timestamp of the last execution, or `null` if never run.</ResponseField>
    <ResponseField name="items[].minIntervalSeconds" type="number">Minimum seconds between executions.</ResponseField>
    <ResponseField name="items[].pausedReason" type="string | null">Reason the flow was paused, or `null` if active.</ResponseField>
    <ResponseField name="items[].templateSlug" type="string | null">Template used to create the flow, or `null`.</ResponseField>
    <ResponseField name="items[].xAccountId" type="string | null">Linked X account ID, or `null`.</ResponseField>
    <ResponseField name="items[].createdAt" type="string">ISO 8601 creation timestamp.</ResponseField>
    <ResponseField name="items[].updatedAt" type="string">ISO 8601 last-updated timestamp.</ResponseField>
