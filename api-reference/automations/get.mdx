## Path Parameters

<ParamField path="slug" type="string" required>
  The flow slug. Returned when you [create a flow](/api-reference/automations/create) or [list flows](/api-reference/automations/list).
</ParamField>

## Headers

<ParamField header="x-api-key" type="string" required>
  Your API key. Session cookie authentication is also supported. Generate a key from the [dashboard](https://xquik.com/dashboard).
</ParamField>

## Response

<Tabs>
  <Tab title="200 OK">
    Returns the flow object, its steps, and the 20 most recent runs.

    <ResponseField name="id" type="string">Unique flow ID.</ResponseField>
    <ResponseField name="name" type="string">Flow name.</ResponseField>
    <ResponseField name="slug" type="string">URL-safe flow identifier.</ResponseField>
    <ResponseField name="triggerType" type="string">Trigger type.</ResponseField>
    <ResponseField name="triggerConfig" type="object">Trigger configuration.</ResponseField>
    <ResponseField name="isActive" type="boolean">Whether the flow is active.</ResponseField>
    <ResponseField name="runCount" type="string">Total execution count.</ResponseField>
    <ResponseField name="lastRunAt" type="string | null">ISO 8601 timestamp of last execution.</ResponseField>
    <ResponseField name="minIntervalSeconds" type="number">Minimum seconds between executions.</ResponseField>
    <ResponseField name="pausedReason" type="string | null">Reason the flow was paused, or `null`.</ResponseField>
    <ResponseField name="templateSlug" type="string | null">Template used to create the flow.</ResponseField>
    <ResponseField name="xAccountId" type="string | null">Linked X account ID.</ResponseField>
    <ResponseField name="createdAt" type="string">ISO 8601 creation timestamp.</ResponseField>
    <ResponseField name="updatedAt" type="string">ISO 8601 last-updated timestamp.</ResponseField>
    <ResponseField name="steps" type="object[]">Array of step objects. See fields below.</ResponseField>
    <ResponseField name="recentRuns" type="object[]">20 most recent run objects.</ResponseField>

    ### Run Object Fields

    <ResponseField name="recentRuns[].id" type="string">Unique run ID.</ResponseField>
    <ResponseField name="recentRuns[].status" type="string">Run status.</ResponseField>
    <ResponseField name="recentRuns[].startedAt" type="string">ISO 8601 start timestamp.</ResponseField>
    <ResponseField name="recentRuns[].completedAt" type="string | null">ISO 8601 completion timestamp.</ResponseField>
    <ResponseField name="recentRuns[].errorMessage" type="string | null">Error message if the run failed, or `null`.</ResponseField>
    <ResponseField name="recentRuns[].triggeredBy" type="string">What triggered the run (e.g., `monitor`, `schedule`, `webhook`, `search`).</ResponseField>

    ### Step Object Fields

    <ResponseField name="steps[].id" type="string">Unique step ID.</ResponseField>
    <ResponseField name="steps[].flowId" type="string">Parent flow ID.</ResponseField>
    <ResponseField name="steps[].stepType" type="string">One of: `action`, `condition`, `extraction`.</ResponseField>
    <ResponseField name="steps[].actionType" type="string | null">Action type if `stepType` is `action`.</ResponseField>
    <ResponseField name="steps[].extractionType" type="string | null">Extraction type if `stepType` is `extraction`.</ResponseField>
    <ResponseField name="steps[].branch" type="string">One of: `main`, `if_true`, `if_false`.</ResponseField>
    <ResponseField name="steps[].config" type="object">Step-specific configuration.</ResponseField>
    <ResponseField name="steps[].position" type="number">Order position in the branch.</ResponseField>
    <ResponseField name="steps[].positionX" type="number | null">Canvas X coordinate.</ResponseField>
    <ResponseField name="steps[].positionY" type="number | null">Canvas Y coordinate.</ResponseField>
    <ResponseField name="steps[].parentStepId" type="string | null">Parent step ID for branching.</ResponseField>
    <ResponseField name="steps[].outputName" type="string | null">Variable name for extraction output.</ResponseField>
    <ResponseField name="steps[].createdAt" type="string">ISO 8601 creation timestamp.</ResponseField>
