  ```go Go
  package main

  import (
      "bytes"
      "encoding/json"
      "fmt"
      "net/http"
  )

  func main() {
      body, _ := json.Marshal(map[string]string{
          "locale": "tr",
      })

      req, err := http.NewRequest("PATCH", "https://xquik.com/api/v1/account", bytes.NewReader(body))
      if err != nil {
          panic(err)
      }
      req.Header.Set("x-api-key", "xq_YOUR_KEY_HERE")
      req.Header.Set("Content-Type", "application/json")

      resp, err := http.DefaultClient.Do(req)
      if err != nil {
          panic(err)
      }
      defer resp.Body.Close()

      var data map[string]interface{}
      if err := json.NewDecoder(resp.Body).Decode(&data); err != nil {
          panic(err)
      }
      fmt.Println(data)
  }
  ```

## Headers

<ParamField header="x-api-key" type="string" required>
  Your API key. Session cookie authentication is also supported. Generate a key from the [dashboard](https://xquik.com/dashboard).
</ParamField>

<ParamField header="Content-Type" type="string" required>
  Must be `application/json`.
</ParamField>

## Body

<ParamField body="locale" type="string" required>
  Locale preference for the account. Must be one of `en`, `tr`, or `es`.
</ParamField>

## Response

<Tabs>
  <Tab title="200 OK">
    <ResponseField name="success" type="boolean">
      Always `true` on successful update.
    </ResponseField>

    ```json
    { "success": true }
    ```
  </Tab>
  <Tab title="400 Invalid Input">
    ```json
    { "error": "invalid_input" }
    ```
    Missing `locale` field or value is not one of `en`, `tr`, `es`.
  </Tab>
  <Tab title="401 Unauthenticated">
    ```json
    { "error": "unauthenticated" }
    ```
    Missing or invalid API key.
  </Tab>
</Tabs>
